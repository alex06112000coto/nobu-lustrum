===================================================================
   FIREBASE SECURITY RULES FOR CLOUD FIRESTORE & FIREBASE STORAGE
===================================================================

Copy and paste these rules into your Firebase Console under the "Rules" 
tabs for Cloud Firestore and Firebase Storage respectively.

-------------------------------------------------------------------
1. CLOUD FIRESTORE SECURITY RULES
-------------------------------------------------------------------

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }

    // Helper function to check if user owns their user document
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // 1. Users collection rules
    match /users/{userId} {
      // Authenticated users can search the list of participants to vote
      allow read: if isAuthenticated();
      // Users can only modify their own profile data (nickname, real name, avatar)
      allow create, update: if isAuthenticated() && isOwner(userId);
      // Nobody can delete user profiles
      allow delete: if false;
    }

    // 2. Photos collection rules (Home Feed)
    match /photos/{photoId} {
      // Authenticated users can read the photo feed and write new posts
      allow read, create: if isAuthenticated();
      
      // Only authenticated users can like or comment on photos
      // Note: We enforce that updates can only modify comments or likes arrays
      allow update: if isAuthenticated() 
        && (request.resource.data.diff(resource.data).affectedKeys().hasAny(['likes', 'comments']));
        
      // Users can only delete their own photos
      allow delete: if isAuthenticated() && resource.data.uid == request.auth.uid;
    }

    // 3. Daily 'Who is most likely to' votes collection
    match /likely_votes/{dateKey} {
      // Logged in users can read today's vote document
      allow read: if isAuthenticated();
      
      // Logged in users can create/update today's votes document to cast their vote or write a comment
      allow create, update: if isAuthenticated();
      
      // Nobody can delete voting tallies
      allow delete: if false;
    }

    // 4. Invite Code Config rules
    match /config/invite {
      // Allow unauthenticated reads for signup validation, disable writes
      allow read: if true;
      allow write: if false;
    }
  }
}

2. FIREBASE STORAGE SECURITY RULES
-------------------------------------------------------------------

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Helper to check authentication
    function isAuthenticated() {
      return request.auth != null;
    }

    // Helper to verify matching owner
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // 1. Profile Avatars (Stored under /avatars/uid)
    match /avatars/{userId} {
      // Anyone logged in can view participants' profile pictures
      allow read: if isAuthenticated();
      // Users can only upload, overwrite or delete their own profile avatar
      // We enforce size constraints to prevent DoS (max 3MB for avatar images)
      allow write: if isAuthenticated() 
        && isOwner(userId) 
        && request.resource.size < 3 * 1024 * 1024
        && request.resource.contentType.matches('image/.*');
    }

    // 2. Feed Photos (Stored under /feed_photos/randomUuid)
    match /feed_photos/{photoId} {
      // Logged-in participants can view and upload photos to the feed
      allow read: if isAuthenticated();
      // We enforce file type validation and size limits (max 5MB for feed uploads)
      allow create: if isAuthenticated() 
        && request.resource.size < 5 * 1024 * 1024
        && request.resource.contentType.matches('image/.*');
        
      // Nobody can overwrite an existing photo post once uploaded
      allow update: if false;
      
      // Users can delete files (which they own) - handled in combination with database logic
      allow delete: if isAuthenticated();
    }
  }
}
